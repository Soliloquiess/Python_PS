import java.util.*;
class Main{
	int[][] dy=new int[35][35];
	public int DFS(int n, int r){
		if(dy[n][r]>0) return dy[n][r];	//메모이제이션 사용 이미 구한값이 있으면 그 값을 리턴한다.
		if(n==r || r==0) return 1;	//n과 r이 같던지 
		else return dy[n][r]=DFS(n-1, r-1)+DFS(n-1, r);	//공식 사용
	}
	public static void main(String[] args){
		Main T = new Main();
		Scanner kb = new Scanner(System.in);
		int n=kb.nextInt();
		int r=kb.nextInt();
		System.out.println(T.DFS(n, r));
	}
}

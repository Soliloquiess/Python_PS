import java.util.*;
class Main{
	static int n, m, answer=Integer.MAX_VALUE;
//결국 큰금액 부터 찾고 레벨을 커트하는 것
	public void DFS(int L, int sum, Integer[] arr){
		if(sum>m) return;	//m을 넘어가면 뻗지 못하도록 해준다.
		if(L>=answer) return;	//L이 answer보다 크거나 갇아도 리턴시키면 된다.
		if(sum==m){
			answer=Math.min(answer, L);	//sum을 만들어내는 개수가 L
		}
		else{
			for(int i=0; i<n; i++){	
				DFS(L+1, sum+arr[i], arr);	//i번쨰 동전 하나 사용
			}
		}	
	}
	public static void main(String[] args){
		Main T = new Main();
		Scanner kb = new Scanner(System.in);
		n=kb.nextInt();
		Integer[] arr=new Integer[n];
		for(int i=0; i<n; i++) arr[i]=kb.nextInt();
		Arrays.sort(arr, Collections.reverseOrder());	//reverseOrder로 내림차순 해준다.
		m=kb.nextInt();
		T.DFS(0, 0, arr);
		System.out.println(answer);
	}
}

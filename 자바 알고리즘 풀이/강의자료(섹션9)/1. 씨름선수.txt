import java.util.*;
class Body implements Comparable<Body>{
    public int h, w;
    Body(int h, int w) {
        this.h = h;
        this.w = w;
    }
    @Override
    public int compareTo(Body o){
        return o.h-this.h;	//h로 기준 잡고 내림차순
    }
}

class Main {
	public int solution(ArrayList<Body> arr, int n){
		int cnt=0;
		Collections.sort(arr);	//컬렉션 소트(위에서 설정한 대로 내림차순)
		int max=Integer.MIN_VALUE;	(가장 작은값)
		for(Body ob : arr){	//위에서 정렬했으니까 몸무게 순으로 정렬하면 된다.
			if(ob.w>max){	//ob의 몸무게가 최대보다 크면 카운트 추가
				max=ob.w;
				cnt++;
			}
		}
		return cnt;
	}

	public static void main(String[] args){
		Main T = new Main();
		Scanner kb = new Scanner(System.in);
		int n=kb.nextInt();
		ArrayList<Body> arr = new ArrayList<>();
		for(int i=0; i<n; i++){
			int h=kb.nextInt();
			int w=kb.nextInt();
			arr.add(new Body(h, w));	//바디객체에 넣음
		}
		System.out.println(T.solution(arr, n));
	}
}

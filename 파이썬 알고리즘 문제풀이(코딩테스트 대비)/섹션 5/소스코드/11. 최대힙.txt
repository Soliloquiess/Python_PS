import sys
import heapq as hq
sys.stdin=open("input.txt", "r")
a=[]

#앞의 최소힙 코드에서 조금만 변경하면 됨.
#입력할 때 부호 바꿔버리면 됨.
while True:
    n=int(input())
    if n==-1:
        break
    if n==0:
        if len(a)==0:
            print(-1)
        else:
            print(-hq.heappop(a))
    else:
        hq.heappush(a, -n)
